Supabase CRUD 测试

首次使用：点击展开建表 SQL

在 Supabase 控制台 → SQL Editor 中执行以下 SQL：

CREATE TABLE todos (
  id BIGSERIAL PRIMARY KEY,
  text TEXT NOT NULL,
  done BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

ALTER TABLE todos ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Public read" ON todos FOR SELECT USING (true);
CREATE POLICY "Public insert" ON todos FOR INSERT WITH CHECK (true);
CREATE POLICY "Public update" ON todos FOR UPDATE USING (true);
CREATE POLICY "Public delete" ON todos FOR DELETE USING (true);
